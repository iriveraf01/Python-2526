# Filtrar Espaa y total

    (data['Location'] == 'Spain') &
    (data['Dim1'] == 'Both sexes')
]

    spain_data['Period'],

)

ax.set_title('Evolucin de la esperanza de vida en Espaa')

ax.legend()
ax.grid(True)

plt.tight_layout()
plt.show()

import pandas as pd
import matplotlib.pyplot as plt

# Cargar los datos del fichero CSV
data = pd.read_csv('paises.csv')

# Filtrar solo los pases de Europa
europa = data[data['continente'] == 'Europe']

# Ordenar por PIB de mayor a menor
europa_top10 = europa.sort_values(by='PIB', ascending=False).head(10)

# Crear el grfico de tarta
plt.figure(figsize=(8, 8))

plt.pie(
    europa_top10['PIB'],              # Valores numricos (PIB)
    labels=europa_top10['pais'],      # Etiquetas de cada porcin
    autopct='%1.1f%%',                # Muestra el porcentaje con 1 decimal
    startangle=90                     # Gira el grfico para mejor visualizacin
)

# Ttulo del grfico
plt.title('Distribucin del PIB en Europa (Top 10 pases)')

# Hace que el grfico sea un crculo perfecto
plt.axis('equal')

plt.show()

import pandas as pd
import matplotlib.pyplot as plt

# Cargar los datos
paises = pd.read_csv('paises.csv')
medicos = pd.read_csv('medicos.csv')

# Renombrar columnas para facilitar el trabajo
medicos = medicos.rename(columns={
    'Location': 'pais',
    'First Tooltip': 'medicos'
})

# Unin interna entre pases y mdicos
datos = pd.merge(
    paises,
    medicos,
    on='pais',
    how='inner'
)

# Agrupar por regin y calcular la media de PIB y mdicos
datos_region = (
    datos
    .groupby('region', as_index=False)
    .agg({
        'PIB': 'mean',
        'medicos': 'mean'
    })
)

# Calcular la media mundial
media_pib = datos_region['PIB'].mean()
media_medicos = datos_region['medicos'].mean()

# Crear la figura
plt.figure(figsize=(10, 6))

# Nube de puntos por regin
plt.scatter(
    datos_region['PIB'],
    datos_region['medicos'],
    label='Regiones'
)

# Punto medio (media mundial)
plt.scatter(
    media_pib,
    media_medicos,
    color='red',
    s=120,
    label='Media mundial'
)

# Ttulos y etiquetas
plt.title('Relacin entre el PIB y el nmero de mdicos por regin')
plt.xlabel('PIB')
plt.ylabel('Nmero de mdicos')

# Leyenda y rejilla
plt.legend()
plt.grid(True)

plt.tight_layout()
plt.show()

Grfica 5.

Haz lo mismo que en grfica 4 pero desglosado por continente.

Para ello debemos crear la grfica a partir de subplots. Cada continente es un subplot diferente, debemos hacer 2 columnas y 3 filas, cada continente es un subplot.

In[6]:
import pandas as pd
import matplotlib.pyplot as plt

# Cargar los datos
paises = pd.read_csv('paises.csv')
medicos = pd.read_csv('medicos.csv')

# Renombrar columnas del fichero de mdicos
medicos = medicos.rename(columns={
    'Location': 'pais',
    'First Tooltip': 'medicos'
})

# Unin interna
datos = pd.merge(
    paises,
    medicos,
    on='pais',
    how='inner'
)

# Obtener lista de continentes
continentes = datos['continente'].unique()

# Crear la figura con 3 filas y 2 columnas de subplots
fig, axes = plt.subplots(3, 2, figsize=(14, 12))
axes = axes.flatten()  # Facilita recorrerlos en un bucle

# Recorrer cada continente y su subplot
for i, continente in enumerate(continentes):
    ax = axes[i]

    # Filtrar datos del continente
    datos_cont = datos[datos['continente'] == continente]

    # Agrupar por regin
    datos_region = (
        datos_cont
        .groupby('region', as_index=False)
        .agg({
            'PIB': 'mean',
            'medicos': 'mean'
        })
    )

    # Calcular medias del continente
    media_pib = datos_region['PIB'].mean()
    media_medicos = datos_region['medicos'].mean()

    # Nube de puntos
    ax.scatter(
        datos_region['PIB'],
        datos_region['medicos'],
        label='Regiones'
    )

    # Punto medio del continente
    ax.scatter(
        media_pib,
        media_medicos,
        color='red',
        s=80,
        label='Media del continente'
    )

    # Ttulo y etiquetas
    ax.set_title(continente)
    ax.set_xlabel('PIB')
    ax.set_ylabel('Nmero de mdicos')

    # Rejilla y leyenda
    ax.grid(True)
    ax.legend(fontsize=8)

# Elimina subplots vacos si hay menos de 6 continentes
for j in range(i + 1, len(axes)):
    fig.delaxes(axes[j])

# Ajustar espaciado
plt.tight_layout()
plt.show()

import pandas as pd
import matplotlib.pyplot as plt

# Cargar los datos
paises = pd.read_csv('paises.csv')
medicos = pd.read_csv('medicos.csv')

# Renombrar columnas del fichero de mdicos
medicos = medicos.rename(columns={
    'Location': 'pais',
    'First Tooltip': 'medicos'
})

# Unin interna entre ambos CSV
datos = pd.merge(
    paises,
    medicos,
    on='pais',
    how='inner'
)

# Redondear la cantidad de mdicos al entero ms cercano
# round() redondea al entero ms prximo
datos['medicos_redondeados'] = datos['medicos'].round().astype(int)

# Crear el histograma
plt.figure(figsize=(10, 6))

plt.hist(
    datos['medicos_redondeados'],  # Datos del eje X
    bins=15,                       # Nmero de intervalos del histograma
    edgecolor='black'              # Bordes para distinguir mejor las barras
)

# Ttulos y etiquetas
plt.title('Distribucin de la cantidad de mdicos por pas')
plt.xlabel('Cantidad de mdicos (redondeada)')
plt.ylabel('Nmero de pases')

# Rejilla para facilitar la lectura
plt.grid(axis='y')

plt.tight_layout()
plt.show()
